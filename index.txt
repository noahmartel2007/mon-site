      <script>
       function actualiserHeureEtDate() {
           
            const zone = { timeZone: 'America/Martinique' };
 
            const optionsDate = { 
                ...zone, 
                weekday: 'long', 
                day: 'numeric', 
                month: 'long', 
                year: 'numeric' 
            };
            
            const optionsHeure = { 
                ...zone, 
                hour: '2-digit', 
                minute: '2-digit', 
                second: '2-digit' 
            };
            const maintenant = new Date();
            
            const dateFr = maintenant.toLocaleDateString('fr-FR', optionsDate);
            const heureFr = maintenant.toLocaleTimeString('fr-FR', optionsHeure);
        
            document.getElementById("demo").innerHTML = dateFr + " - " + heureFr;
        }
        setInterval(actualiserHeureEtDate, 1000);

        actualiserHeureEtDate();
      </script>
