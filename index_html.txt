<body class="container-fluid">

    <!-- LISTE -->
    <div id="vue-liste">
        <h1>Liste des étudiants</h1>

        <table class="table table-striped table-hover">
            <thead>
                <tr>
                    <th>ID</th>
                    <th>Nom</th>
                    <th>Prénom</th>
                    <th>Email</th>
                </tr>
            </thead>
            <tbody id="tableau"></tbody>
        </table>

        <!-- BOUTON AJOUT EN BAS -->
        <div class="mt-3">
            <button class="btn btn-primary" onclick="ajouterMode()">
                + Ajouter un étudiant
            </button>
        </div>
    </div>

    <!-- FORMULAIRE -->
    <div id="vue-form" style="display: none;">
        <h1 id="titreForm">Détails et Modification</h1>

        <form class="student">
            <div class="mb-3">
                <label>Nom</label>
                <input type="text" id="nom" class="form-control">
            </div>

            <div class="mb-3">
                <label>Prénom</label>
                <input type="text" id="prenom" class="form-control">
            </div>

            <div class="mb-3">
                <label>Email</label>
                <input type="email" id="email" class="form-control">
            </div>

            <div class="mt-4">
                <button type="button" class="btn btn-primary" onclick="retourListe()">Retour</button>
                <button type="button" class="btn btn-warning" onclick="sauvegarder()">Enregistrer</button>
                <button type="button" class="btn btn-danger" onclick="supprimerEtudiant()" id="btnDelete">Supprimer</button>
            </div>
        </form>
    </div>

<script>
var idActuel = null;
var mode = "ajout";
var apiBase = "http://192.168.1.224/ciel2027/api";

document.addEventListener('DOMContentLoaded', chargerListe);

// LISTE
function chargerListe() {
    var req = new XMLHttpRequest();
    req.open("GET", apiBase + "/etudiants", true);
    req.send();

    req.onload = function() {
        var json = JSON.parse(req.responseText);
        var html = "";

        json.forEach(function(val) {
            html += "<tr>";
            html += "<td>" + val.id + "</td>";
            html += "<td><a href='#' onclick='voirDetails(" + val.id + ")'>" + val.nom + "</a></td>";
            html += "<td>" + val.prenom + "</td>";
            html += "<td>" + val.email + "</td>";
            html += "</tr>";
        });

        document.getElementById('tableau').innerHTML = html;
    };
}

// VOIR / MODIFIER
function voirDetails(id) {
    mode = "modif";
    idActuel = id;

    var req = new XMLHttpRequest();
    req.open("GET", apiBase + "/etudiant/" + id, true);
    req.send();

    req.onload = function () {
        var json = JSON.parse(req.responseText);

        document.getElementById("nom").value = json.nom;
        document.getElementById("prenom").value = json.prenom;
        document.getElementById("email").value = json.email;

        document.getElementById("titreForm").innerText = "Modifier un étudiant";
        document.getElementById("btnDelete").style.display = "inline-block";

        afficherForm();
    };
}

// AJOUT
function ajouterMode() {
    mode = "ajout";
    idActuel = null;

    document.getElementById("nom").value = "";
    document.getElementById("prenom").value = "";
    document.getElementById("email").value = "";

    document.getElementById("titreForm").innerText = "Ajouter un étudiant";
    document.getElementById("btnDelete").style.display = "none";

    afficherForm();
}

// SAVE
function sauvegarder() {
    var data = {
        nom: document.getElementById("nom").value,
        prenom: document.getElementById("prenom").value,
        email: document.getElementById("email").value
    };

    var req = new XMLHttpRequest();

    if (mode === "ajout") {
        req.open("POST", apiBase + "/etudiants", true);
    } else {
        req.open("PUT", apiBase + "/etudiant/" + idActuel, true);
    }

    req.setRequestHeader('Content-Type', 'application/json');
    req.send(JSON.stringify(data));

    req.onload = function () {
        retourListe();
        chargerListe();
    };
}

// DELETE
function supprimerEtudiant() {
    if (!confirm("Supprimer cet étudiant ?")) return;

    var req = new XMLHttpRequest();
    req.open("DELETE", apiBase + "/etudiant/" + idActuel, true);
    req.send();

    req.onload = function () {
        retourListe();
        chargerListe();
    };
}

// NAVIGATION
function afficherForm() {
    document.getElementById('vue-liste').style.display = 'none';
    document.getElementById('vue-form').style.display = 'block';
}

function retourListe() {
    document.getElementById('vue-liste').style.display = 'block';
    document.getElementById('vue-form').style.display = 'none';
}
</script>
